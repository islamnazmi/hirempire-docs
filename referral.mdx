<script>
document.addEventListener('DOMContentLoaded', function() {
  const emailInput = document.getElementById('referral-email');
  const button = document.getElementById('referral-button');
  const resultDiv = document.getElementById('referral-result');
  
  button.addEventListener('click', async function() {
    const email = emailInput.value.trim();
    
    if (!email) {
      resultDiv.innerHTML = '<div class="error-message">Please enter your email address</div>';
      return;
    }
    
    if (!isValidEmail(email)) {
      resultDiv.innerHTML = '<div class="error-message">Please enter a valid email address</div>';
      return;
    }
    
    button.disabled = true;
    button.textContent = 'Loading...';
    resultDiv.innerHTML = '';
    
    try {
      const response = await fetch('https://api.hirempire.com/v1/mint-refer', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email })
      });
      
      if (!response.ok) {
        throw new Error('Request failed. Please try again.');
      }
      
      const data = await response.json();
      resultDiv.innerHTML = `<div class="success-message">üéâ Your referral link: ${data.link}</div>`;
      
    } catch (error) {
      resultDiv.innerHTML = `<div class="error-message">${error.message}</div>`;
    }
    
    button.disabled = false;
    button.textContent = 'Get Referral Link';
  });
  
  function isValidEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  }
});
</script>

<script>
function toggleAccordion(button) {
  const content = button.nextElementSibling;
  const icon = button.querySelector('span');
  
  if (content.classList.contains('active')) {
    content.classList.remove('active');
    icon.textContent = '+';
  } else {
    content.classList.add('active');
    icon.textContent = '-';
  }
}
